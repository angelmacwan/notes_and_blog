Error Handling — try / except / finally

Errors (called exceptions) crash your program. Wrap risky code in a try block so you can handle errors gracefully.

try:
    age = int("abc")

def safe_divide(a, b):
    try:
        result = a / b
        return result
    except ZeroDivisionError:
        print("  Error: Can't divide by zero!")
    except TypeError:
        print("  Error: Both values must be numbers!")
    finally:
        print("  (divide attempt finished)")
    return None

print(safe_divide(10, 2))
print(safe_divide(10, 0))
print(safe_divide(10, "x"))

Classes and Object-Oriented Programming

A class is a blueprint for creating objects. Each object (instance) has its own data (attributes) and can perform actions (methods).

class Animal:

    def __init__(self, name, species, age):
        self.name    = name
        self.species = species
        self.age     = age
        self.energy  = 100

    def eat(self, food):
        self.energy += 10
        print(f"{self.name} eats {food}. Energy: {self.energy}")

    def run(self):
        if self.energy >= 20:
            self.energy -= 20
            print(f"{self.name} runs! Energy: {self.energy}")
        else:
            print(f"{self.name} is too tired to run.")

    def __str__(self):
        return f"{self.name} ({self.species}, age {self.age})"

dog = Animal("Rex", "Dog", 3)
cat = Animal("Whiskers", "Cat", 5)

print(dog)
dog.eat("a bone")
dog.run()
dog.run()

Rex runs! Energy: 90
Rex runs! Energy: 70

Inheritance

A child class inherits all attributes and methods from a parent class, and can add or override them.

class Dog(Animal):

    def __init__(self, name, age, breed):
        super().__init__(name, "Dog", age)
        self.breed  = breed
        self.tricks = []

    def learn_trick(self, trick):
        self.tricks.append(trick)
        print(f"{self.name} learnt '{trick}'!")

    def perform(self):
        if self.tricks:
            import random
            trick = random.choice(self.tricks)
            print(f"{self.name} performs: {trick}!")
        else:
            print(f"{self.name} doesn't know any tricks yet.")

    def __str__(self):
        return f"{self.name} the {self.breed} (age {self.age})"

rex = Dog("Rex", 3, "Labrador")
print(rex)
rex.eat("kibble")
rex.learn_trick("sit")
rex.learn_trick("shake paw")
rex.perform()
Rex the Labrador (age 3)
Rex eats kibble. Energy: 110
Rex learnt 'sit'!
Rex learnt 'shake paw'!
Rex performs: shake paw!

Decorators — a Quick Look

A decorator is a function that wraps another function to add extra behaviour without changing the original code. They use the @ symbol.

import time

def timer(func):
    def wrapper(*args, **kwargs):
        start  = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"  [{func.__name__} took {elapsed:.4f}s]")
        return result
    return wrapper

@timer
def slow_sum(n):
    total = 0
    for i in range(n):
        total += i
    return total

result = slow_sum(1_000_000)
print(f"Sum = {result}")
  [slow_sum took 0.0312s]
Sum = 499999500000

Final Project — Multi-Category Quiz App

This quiz app demonstrates lists, dicts, functions, classes, error handling, and random — everything covered above!

import random

QUESTIONS = {
    "Python": [
        {"q": "What does len([1,2,3]) return?",
         "options": ["A) 2", "B) 3", "C) 4", "D) 1"], "answer": "B"},
        {"q": "Which keyword defines a function?",
         "options": ["A) func", "B) define", "C) def", "D) function"], "answer": "C"},
        {"q": "What is 5 // 2?",
         "options": ["A) 2.5", "B) 3", "C) 2", "D) 1"], "answer": "C"},
    ],
    "Maths": [
        {"q": "What is 7 squared?",
         "options": ["A) 14", "B) 49", "C) 42", "D) 56"], "answer": "B"},
        {"q": "Square root of 144?",
         "options": ["A) 11", "B) 13", "C) 14", "D) 12"], "answer": "D"},
    ],
    "General": [
        {"q": "Capital of France?",
         "options": ["A) Berlin", "B) Madrid", "C) Paris", "D) Rome"], "answer": "C"},
        {"q": "Colours in a rainbow?",
         "options": ["A) 5", "B) 6", "C) 7", "D) 8"], "answer": "C"},
    ],
}

class QuizApp:
    def __init__(self):
        self.score = 0
        self.total = 0
        self.wrong = []

    def pick_category(self):
        cats = list(QUESTIONS.keys()) + ["Random Mix"]
        print("\nCategories:")
        for i, c in enumerate(cats, 1):
            print(f"  {i}. {c}")
        while True:
            try:
                choice = int(input("Pick a category: "))
                if 1 <= choice <= len(cats):
                    cat = cats[choice - 1]
                    if cat == "Random Mix":
                        all_qs = [q for qs in QUESTIONS.values() for q in qs]
                        return random.sample(all_qs, min(5, len(all_qs)))
                    return list(QUESTIONS[cat])
            except ValueError:
                pass
            print("  Invalid.")

    def ask(self, item, num):
        print(f"\nQ{num}: {item['q']}")
        for opt in item["options"]:
            print(f"    {opt}")
        answer = input("Your answer (A/B/C/D): ").strip().upper()
        self.total += 1
        if answer == item["answer"]:
            print("  Correct!")
            self.score += 1
        else:
            print(f"  Wrong! Answer was {item['answer']}.")
            self.wrong.append(item)

    def run(self):
        print("=" * 45)
        print("         PYTHON QUIZ APP")
        print("=" * 45)
        name = input("Enter your name: ").strip() or "Player"

        while True:
            questions = self.pick_category()
            random.shuffle(questions)
            for i, q in enumerate(questions, 1):
                self.ask(q, i)

            pct = (self.score / self.total * 100) if self.total else 0
            print(f"\n--- Results for {name} ---")
            print(f"Score: {self.score}/{self.total}  ({pct:.0f}%)")

            if self.wrong:
                print("\nReview — questions you got wrong:")
                for item in self.wrong:
                    print(f"  Q: {item['q']}  (Answer: {item['answer']})")
                self.wrong.clear()

            again = input("\nPlay again? (y/n): ").strip().lower()
            if again != "y":
                print(f"\nFinal: {self.score}/{self.total}. Thanks, {name}!")
                break

QuizApp().run()
